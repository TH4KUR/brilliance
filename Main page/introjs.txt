
      
let btn_exit = document.getElementById('close_intro')
btn_exit.addEventListener('click', closeModal(document.getElementById('introModal')))
var slideIndex = 1;
showDivs(slideIndex);

function plusDivs(n) {
  showDivs(slideIndex += n);
}

function showDivs(n) {
  var i;
  var x = document.getElementsByClassName("mySlides");
  if (n > x.length) {slideIndex = 1}
  if(x[slideIndex-1].classList.contains('last_slide')){
    btn_exit.classList.remove('hidden')
  }
  if (n < 1) {slideIndex = x.length} ;
  for (i = 0; i < x.length; i++) {
    x[i].style.display = "none";
  }
  x[slideIndex-1].style.display = "flex";
}
